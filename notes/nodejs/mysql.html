    constructor(data, msg = 'æ“ä½œæˆåŠŸ', options) {
        this.data = null
        if (arguments.length === 0) {

            code: this.code,

        this.code = CODE_SUCCESS
        this.json(res)
    }

    fail(res) {
        this.code = CODE_ERROR
        this.json(res)
    }

    expired(res) {
        this.code = CODE_TOKEN_EXPIRED
        this.json(res)
    }
}
