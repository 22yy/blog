var findBottomLeftValue = function(root) {
  if(!root) return;
  let queue=[root];
  let result;
  while(queue.length) {
      let size = queue.length;
      for(let i=0;i<size;i++) {
          let node = queue.shift();
          if(i===0) result = node.val;
          node.left && queue.push(node.left);
          node.right && queue.push(node.right);
      }
  }
  return result;
};
